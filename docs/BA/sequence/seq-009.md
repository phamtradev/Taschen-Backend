    NS-->>AD: "Notification sent successfully"
    PS-->>PC: PromotionResponse
    PC-->>AD: 200 OK + PromotionResponse

    AD->>PC: POST /api/promotions/{id}/notify-paused
    PC->>PS: notifyPromotionPaused(id)
    PS->>UR: findActiveUsers()
    UR->>DB: SELECT * FROM users WHERE is_active = true
    DB-->>UR: List<User>
    UR-->>PS: List<User>
    loop For each user
        PS->>NS: sendNotification(user, "Promotion paused")
        NS-->>PS: done
    end
    PS-->>PC: void
    PC-->>AD: 200 OK
