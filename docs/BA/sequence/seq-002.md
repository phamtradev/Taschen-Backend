    G->>AC: POST /api/auth/login
    AC->>AS: login(AuthenticationRequest)
    AS->>UR: findByEmail(email)
    UR->>DB: SELECT * FROM users WHERE email = ?
    DB-->>UR: User (or null)
    alt User exists
        UR-->>AS: User
        AS->>AS: Verify password with BCrypt
        alt Password correct AND user.isActive == true
            AS->>JS: generateAccessToken(user)
            JS-->>AS: accessToken (JWT)
            AS->>AS: Generate refreshToken
            AS->>RTR: save(refreshToken)
            RTR->>DB: INSERT INTO refresh_tokens (...)
            DB-->>RTR: RefreshToken
            AS-->>AC: AuthenticationResponse
            AC-->>G: 200 OK + AuthenticationResponse
        else User not active (email not verified)
            AS-->>AC: IdInvalidException("Please verify your email")
            AC-->>G: 403 Forbidden
        else Password incorrect
            AS-->>AC: IdInvalidException("Invalid credentials")
            AC-->>G: 401 Unauthorized
        end
    else User not found
        UR-->>AS: null
        AS-->>AC: IdInvalidException("Invalid credentials")
        AC-->>G: 401 Unauthorized
    end
