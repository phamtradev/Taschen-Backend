    DB-->>OS: Order
    OS->>DB: SELECT * FROM order_details WHERE order_id = ?
    DB-->>OS: List<OrderDetail>
    loop For each OrderDetail
        OS->>DB: SELECT * FROM batch_details WHERE order_detail_id = ?
        DB-->>OS: BatchDetail
        OS->>DB: UPDATE batches SET remaining_quantity = remaining_quantity + ?<br/>WHERE id = ?
    end
    OS->>DB: UPDATE orders SET status='CANCELLED' WHERE id = ?
    OS->>NS: notifyCustomer(orderId, "Order cancelled")
    NS-->>U: 🔔 Order cancelled!
