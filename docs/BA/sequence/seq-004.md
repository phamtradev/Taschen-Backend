        OS->>DB: Get OrderDetails
        OS->>BS: restoreStock(orderDetails)
        BS->>DB: UPDATE batches SET remaining_quantity = remaining_quantity + qty WHERE id = ?
        OS->>DB: UPDATE orders SET status = 'CANCELLED' WHERE id = ?
        OS->>NS: sendNotification(userId, "Order cancelled")
        NS-->>U: 🔔 Notification
        OS-->>OC: OrderResponse
        OC-->>U: 200 OK + OrderResponse

        OS-->>OC: IdInvalidException("Cannot cancel order in current status")
